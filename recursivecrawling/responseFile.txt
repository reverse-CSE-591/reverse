  function stateManager() {
    if (http_request.readyState == 4) {
      if (http_request.status == 200) {
        updatepage(http_request.responseText);

        alert('There was a problem with the request.');
      }
    }
  }

    document.getElementById("userlist").innerHTML = str;
  }
